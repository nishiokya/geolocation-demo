🌍 Geolocation API Tutorial

このチュートリアルについて:
ブラウザのGeolocation APIを使用して、現在の位置情報を取得し、 その結果をJSON形式で表示します。
※位置情報の使用許可が求められます。
📝 サンプルコード 
// 現在の位置情報を取得
navigator.geolocation.getCurrentPosition(
    // 成功時のコールバック
    function(position) {
        // toJSON()メソッドで位置情報を取得
        const positionData = position.toJSON();
        console.log('位置情報:', positionData);
        
        // 取得できるデータ
        console.log('緯度:', positionData.coords.latitude);
        console.log('経度:', positionData.coords.longitude);
        console.log('精度:', positionData.coords.accuracy);
    },
    // エラー時のコールバック
    function(error) {
        console.error('エラー:', error.message);
    },
    // オプション
    {
        enableHighAccuracy: true,  // 高精度モード
        timeout: 10000,              // タイムアウト(ms)
        maximumAge: 0                // キャッシュを使用しない
    }
);
