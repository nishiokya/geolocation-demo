🌍 Geolocation API - Watch Position Tutorial

このチュートリアルについて:
ブラウザのGeolocation APIのwatchPosition()を使用して、 位置情報の変化をリアルタイムで監視します。
※位置情報の使用許可が求められます。
📝 サンプルコード 
// 位置監視を開始
let watchId = navigator.geolocation.watchPosition(
    // 成功時のコールバック
    function(position) {
        // toJSON()メソッドで位置情報を取得
        const positionData = position.toJSON();
        console.log('位置情報:', positionData);
        
        // タイムスタンプを日時に変換
        const date = new Date(positionData.timestamp);
        console.log('取得時刻:', date.toLocaleString('ja-JP'));
    },
    // エラー時のコールバック
    function(error) {
        console.error('エラー:', error.message);
    },
    // オプション
    {
        enableHighAccuracy: true,  // 高精度モード
        timeout: 10000,              // タイムアウト(ms)
        maximumAge: 0                // キャッシュを使用しない
    }
);

// 位置監視を停止
navigator.geolocation.clearWatch(watchId);
